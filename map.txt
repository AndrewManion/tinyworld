    <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>
    <script>
        var map = L.map('map').setView([20, 0], 2); // Center the map

        // Add OpenStreetMap tile layer
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '&copy; OpenStreetMap contributors'
        }).addTo(map);

        // Function to add a marker
        function addMarker(lat, lon, title, link) {
            L.marker([lat, lon]).addTo(map)
                .bindPopup(`<b>${title}</b><br><a href="${link}">View More</a>`);
        }

        // Add markers for different locations
        addMarker(48.8584, 2.2945, "Eiffel Tower", "photos.html#eiffel");
        addMarker(40.6892, -74.0445, "Statue of Liberty", "photos.html#liberty");
        addMarker(27.1751, 78.0421, "Taj Mahal", "photos.html#taj");

    </script>
